function gulp-command.option (task, flags, description)

  • description and source-code
    function Option(task, flags, description) {
  this.task = task || null;
  this.flags = flags;
  this.required = ~flags.indexOf('<');
  this.optional = ~flags.indexOf('[');
  this.bool = !~flags.indexOf('-no-');
  flags = flags.split(/[ ,|]+/);
  if (flags.length > 1 && !/^[[<]/.test(flags[1])) this.short = flags.shift();
  this.long = flags.shift();
  this.description = description || '';
  this.shortName = this.short.split('-')[1]
  this.longName = this.long.split('--')[1]
}

Her is an example of how to use gulp-command

```javascript
var gulp        = require('gulp');
var gulpCommand = require('gulp-command')(gulp);

gulp
   .option('<related-task>', '-f, --flag', 'Description
', 'callback')

```

- Related-Task: optional

- -f, --flag: required (both short and long flag are required comma Separated);
- description: required
